#include <stdio.h>

 

int findX (char *s){

 

        int j;

 

        for (j = 0; s[j] != '\0'; j++)

                if (s[j] == 'x')

                        return j;

 

        return -1;

}

 

int main(){

 

        char *str = "This is my string x 2";

 

        printf ("%d\n", findX(str));

 

        return 0;

}
